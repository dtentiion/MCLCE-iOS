  ${UPSTREAM_DIR}/Minecraft.Client/Common
  ${UPSTREAM_DIR}/Minecraft.Client/Common/GameRules
  ${UPSTREAM_DIR}/Minecraft.Client/Common/DLC
  ${UPSTREAM_DIR}/Minecraft.Client/Common/Network
  ${UPSTREAM_DIR}/Minecraft.Client/Common/Audio

  ${CMAKE_SOURCE_DIR}/Minecraft.Client/iOS/compat_headers

# Force-include the iOS PCH-equivalent at the top of every TU. iOS_stdafx.h
# pulls in the C++ standard headers upstream's stdafx.h chain assumes plus
# iOS_WinCompat.h's Win32 type aliases, so upstream files that start with
# `#include "stdafx.h"` resolve cleanly without us having to edit them.
target_compile_options(mcle_world_probe PRIVATE
  -include "${CMAKE_SOURCE_DIR}/Minecraft.Client/iOS/iOS_stdafx.h"
  # Upstream uses __declspec(thread) for thread-local storage in a few
  # files (Tag.cpp etc). clang understands the syntax under -fdeclspec
  # and maps it to thread_local for static storage.
  -fdeclspec
)

target_compile_definitions(mcle_world_probe PRIVATE
  _LIB
  _LARGE_WORLDS
  # Skip debug/profiling code paths that need the global `app`
  # (CMinecraftApp instance). PerformanceTimer.cpp gates its body on
  # !_CONTENT_PACKAGE; many other Minecraft.World files do the same.
  # We do not have a real `app` to bind against in the probe so we
  # take the content-only branch.
  _CONTENT_PACKAGE
  # Marker for probe-only stubs in iOS_WinCompat.h. The main app
  # build does not get this define, so it picks up real upstream
  # definitions instead of conflicting stubs.
  MCLE_PROBE_BUILD=1
)

# Pin to C++14. Upstream is 2014-era code that doesn't need anything newer,
# and C++17's std::byte clashes with extraX64.h:14's `typedef unsigned char
# byte;` whenever upstream files do `using namespace std;` (which is most of
# them). The main app target stays on C++17; this only affects the probe
# static lib so there's no ABI-mixing concern at link time.
set_target_properties(mcle_world_probe PROPERTIES
  CXX_STANDARD 14
  CXX_STANDARD_REQUIRED ON
  CXX_EXTENSIONS OFF
)

# ----------------------------------------------------------------------------
# Phase C link-test target. Pulls every TU out of the static archive via
# -force_load so the linker tries to resolve every symbol referenced by the
# probe set. Almost certainly fails to link against the iOS toolchain - the
# undefined-symbol list is the Phase C work item priority queue.
#
# Gated behind ENABLE_WORLD_PROBE_LINK so the regular probe job stays fast.
# The CI driver runs `cmake --build --target mcle_world_probe_link_test` with
# continue-on-error and uploads the log.

option(ENABLE_WORLD_PROBE_LINK "Try to link mcle_world_probe into a tiny exe" OFF)
if(ENABLE_WORLD_PROBE_LINK)
  add_executable(mcle_world_probe_link_test
    probe_link_test.cpp
  )
  target_include_directories(mcle_world_probe_link_test PRIVATE
    ${UPSTREAM_DIR}/Minecraft.World
    ${UPSTREAM_DIR}/Minecraft.World/x64headers
    ${UPSTREAM_DIR}/Minecraft.Client
    ${UPSTREAM_DIR}/Minecraft.Client/Common
    ${UPSTREAM_DIR}/Minecraft.Client/Common/GameRules
    ${UPSTREAM_DIR}/include
    ${CMAKE_SOURCE_DIR}/Minecraft.Client/iOS
    ${CMAKE_SOURCE_DIR}/Minecraft.Client/iOS/compat_headers
  )
  target_compile_options(mcle_world_probe_link_test PRIVATE
    -include "${CMAKE_SOURCE_DIR}/Minecraft.Client/iOS/iOS_stdafx.h"
    -fdeclspec
  )
  target_compile_definitions(mcle_world_probe_link_test PRIVATE
    _LIB _LARGE_WORLDS _CONTENT_PACKAGE MCLE_PROBE_BUILD=1
  )
  set_target_properties(mcle_world_probe_link_test PROPERTIES
    CXX_STANDARD 14
    CXX_STANDARD_REQUIRED ON
    CXX_EXTENSIONS OFF
  )
  target_link_libraries(mcle_world_probe_link_test PRIVATE
    mcle_world_probe
  )
  # Force-load every .o in the archive so undefined symbols actually surface.
  # Without this the linker would only pull TUs that satisfy the test's
  # tiny call set, hiding 99% of the Phase C work list.
  target_link_options(mcle_world_probe_link_test PRIVATE
    "LINKER:-force_load,$<TARGET_FILE:mcle_world_probe>"
  )
endif()
